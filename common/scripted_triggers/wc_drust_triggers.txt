is_from_the_drust_trigger = {
	OR = {
		AND = {
			has_title = title:k_drustvar
			owns_story_of_type = story_drust_incursion
		}
		OR = {
			is_vassal_or_below_of = title:k_drustvar.holder
			employer ?= title:k_drustvar.holder
			is_courtier_of = title:k_drustvar.holder
		}
	}
}

is_from_kul_tiras_trigger = {
