wc_valid_light_faith_for_northshire_trigger = {
	this = faith:cleric
}

brotherhood_have_not_existed_trigger = {
	# Destroyed after the First War
	game_start_date < 587.1.1
	NOR = {
		# And otherwise can't currently be created.
		exists = global_var:brotherhood_of_northshire_title
		# Nor should they have been made via decision or event.
		is_target_in_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:flag_enforced_brotherhood
		}
	}
}

silver_hand_have_not_existed_trigger = {
	OR = {
		game_start_date < 587.1.1
	}
	NOR = {
		exists = global_var:silver_hand_title

		is_target_in_global_variable_list = {
			name = unavailable_unique_decisions
			target = flag:flag_silver_hand_creation
		}
	}
}

is_original_silver_hand_paladin = {
	is_adult = yes
	is_imprisoned = no
	is_incapable = no
	has_religion = religion:tyrism_religion
	OR = {
		is_uther_trigger = yes
		is_turalyon_trigger = yes
		is_saidan_trigger = yes
		is_tirion_trigger = yes
		is_gavinrad_trigger = yes 
	}
}

is_secondary_silver_hand_paladin = {
	is_adult = yes
	is_imprisoned = no
	is_incapable = no
	has_religion = religion:tyrism_religion
	OR = {
		is_alexandros_trigger = yes
		is_maxwell_tyrosus_trigger = yes
		is_raymond_george_trigger = yes 
		is_bolvar_trigger = yes
		is_james_abbendis_trigger = yes
	}
}

is_tertiary_silver_hand_paladin  {
	is_adult = yes
	is_imprisoned = no
	is_incapable = no
	has_religion = religion:tyrism_religion
	OR = {
		is_ballador_the_bright_trigger = yes
		is_sage_truthbearer_trigger = yes
		is_halahk_lifebringer_trigger = yes
		is_magroth_defender_trigger = yes
		is_dagren_orcslayer_trigger = yes
	}
}

can_be_first_paladin_trigger = {
	is_imprisoned = no
	is_incapable = no
	has_religion = religion:tyrism_religion
	
	piety_level >= 1
	prowess >= 15
	learning >= 10
	num_virtuous_traits >= 1
	OR = {
		has_trait = education_martial_3
		has_trait = education_martial_4
		has_trait = education_martial_5
		has_trait = education_learning_3
		has_trait = education_learning_4
		has_trait = education_learning_5
	}
}
