is_known_reagent = {
	is_target_in_variable_list = {
		name = known_reagents
		target = flag:$REAGENT$
	}
}

is_known_recipe = {
	is_target_in_variable_list = {
		name = known_recipes
		target = flag:$RECIPE$
	}
}

has_sufficient_reagent = {
	var:$REAGENT$_amount >= $REQUIRED_AMOUNT$
}

has_profession = {
	OR = {
		var:profession_1 ?= flag:profession_$PROFESSION$
		var:profession_2 ?= flag:profession_$PROFESSION$
	}
}

has_any_profession = {
	NOR = {
		var:profession_1 ?= flag:none
		var:profession_2 ?= flag:none
	}
}
