##MINING REAGENTS 
is_mining_reagent = {
	any_in_global_list = { variable = mining_reagents this = flag:$REAGENT$ }
}

province_has_reagent_trigger = {
	OR = {
		province_has_$REAGENT$_trigger = yes 
		AND = {
			reagent_is_gem_trigger = { REAGENT = $REAGENT$ }
			has_building_with_flag = mine
		}
	}
}
