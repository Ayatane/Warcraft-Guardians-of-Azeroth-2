show_light_gui = {
	scope = character
	is_shown = {
		trigger_if = {
			limit = { is_alive = yes has_variable = light_void_infusion }
			var:light_void_infusion >= 0
		}
		trigger_else = {
			always = yes
		}
	}
}
show_void_gui = {
	scope = character
	is_shown = {
		is_alive = yes 
		has_variable = light_void_infusion
		var:light_void_infusion < 0
	}
}
light_void_max_infusion_gui = {
	scope = character
	is_shown = {
		OR = {
			light_void_infusion_level_value = 3
			light_void_infusion_level_value = -3
		}
	}
}
