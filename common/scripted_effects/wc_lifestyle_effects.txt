set_long_lived_lifestyle_xp_slowdown_event_effect = {
	trigger_event = WCLIF.1
}

set_long_lived_lifestyle_xp_slowdown_delayed_event_effect = {
	trigger_event = { id = WCLIF.1 delayed = yes }
}

set_long_lived_lifestyle_xp_slowdown_effect = {
	if = {
		limit = {
			has_character_modifier = long_lived_lifestyle_xp_slowdown
		}
		remove_character_modifier = long_lived_lifestyle_xp_slowdown
	}
	if = {
		limit = {
			long_lived_lifestyle_xp_slowdown_value > 0
		}
		add_character_modifier = {
			modifier = long_lived_lifestyle_xp_slowdown
			desc = long_lived_lifestyle_xp_slowdown_custom_desc
		}
	}
}
clear_long_lived_lifestyle_xp_slowdown_effect = {
	remove_character_modifier = long_lived_lifestyle_xp_slowdown
}

# used for wc general lifestyle events
# sadly i think this is the only way to do this - SS
add_lifestyle_xp_to_current_focus = { 
	if = { 
		limit = { 
			OR = { 
				has_focus = intrigue_intimidation_focus
				has_focus = intrigue_skulduggery_focus
				has_focus = intrigue_temptation_focus
			}
		}
		add_intrigue_lifestyle_xp = $VALUE$
	}
	else_if = { 
		limit = { 
			OR = { 
				has_focus = martial_authority_focus
				has_focus = martial_chivalry_focus
				has_focus = martial_strategy_focus
			}
		}
		add_martial_lifestyle_xp = $VALUE$
	}
	else_if = { 
		limit = { 
			OR = { 
				has_focus = stewardship_domain_focus
				has_focus = stewardship_duty_focus
				has_focus = stewardship_wealth_focus
			}
		}
		add_stewardship_lifestyle_xp = $VALUE$
	}
	else_if = { 
		limit = { 
			OR = { 
				has_focus = learning_medicine_focus
				has_focus = learning_scholarship_focus
				has_focus = learning_theology_focus
			}
		}
		add_learning_lifestyle_xp = $VALUE$
	}
	else_if = { 
		limit = { 
			OR = { 
				has_focus = diplomacy_family_focus
				has_focus = diplomacy_foreign_affairs_focus
				has_focus = diplomacy_majesty_focus
			}
		}
		add_diplomacy_lifestyle_xp = $VALUE$
	}
	else = { 
		add_xp_to_current_magic_lifestyle_focus = { VALUE = $VALUE$ }
	}
}
