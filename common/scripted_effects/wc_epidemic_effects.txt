copy_epidemic_to_province = { 
	$PROVINCE$ =  { 
		if = { 
			limit = { 
				$EPIDEMIC$ = { 
					has_variable = uniqe_epidemic_loc
				}
			}

			set_variable = { 
				name = uniqe_epidemic_loc
				value = $EPIDEMIC$.var:uniqe_epidemic_loc
			}
		}
	}

	$EPIDEMIC$ = { 
		switch = { # Can't access/copy outbreak_intensity directly, so using switch instead
			trigger = outbreak_intensity
			minor = { 
				$PROVINCE$ = {
					create_epidemic_outbreak = { 
						type = $EPIDEMIC$.epidemic_type
						intensity = minor
						save_scope_as = copied_epidemic
					}
				}
			}
			major = { 
				$PROVINCE$ = {
					create_epidemic_outbreak = { 
						type = $EPIDEMIC$.epidemic_type
						intensity = major
						save_scope_as = copied_epidemic
					}
				}
			}
			apocalyptic = { 
				$PROVINCE$ = {
					create_epidemic_outbreak = { 
						type = $EPIDEMIC$.epidemic_type
						intensity = apocalyptic
						save_scope_as = copied_epidemic
					}
				}
			}
		}
	}
}
