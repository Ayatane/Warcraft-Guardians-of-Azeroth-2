add_magic_infusion_effect = {
	save_scope_value_as = {
		name = infusion_type
		value = flag:$MAGIC$_infusion
	}
	
	save_scope_value_as = {
		name = infusion_value
		value = $ADD$
	}

	custom_description = {
		text = $MAGIC$_infusion_desc
		value = $ADD$
	} 

	if = {
		limit = {
			NOT = {
				has_variable = $MAGIC$_infusion
			}
		}
		set_variable = {
			name = $MAGIC$_infusion
			value = 0
		}
	}
	change_variable = {
		name = $MAGIC$_infusion
		add = $ADD$
	}
}
