light_void_infusion_level_value = {
	value = 0
	if = {
		limit = { is_alive = yes has_variable = light_void_infusion }
		if = {
			limit = { var:light_void_infusion >= 300 }
			add = 3
		}
		else_if = {
			limit = { var:light_void_infusion >= 200 }
			add = 2
		}
		else_if = {
			limit = { var:light_void_infusion >= 100 }
			add = 1
		}
		
		else_if = {
			limit = { var:light_void_infusion <= -300 }
			add = -3
		}
		else_if = {
			limit = { var:light_void_infusion <= -200 }
			add = -2
		}
		else_if = {
			limit = { var:light_void_infusion <= -100 }
			add = -1
		}
	}
}

light_void_infusion_gui_right_frame_value = {
	value = light_void_infusion_gui_left_frame_value
	if = {
		limit = {
			OR = {
				light_void_infusion_level_value = 3
				light_void_infusion_level_value = -3
			}
		}
		value = 0
	}
	else_if = {
		limit = {
			is_alive = yes 
			trigger_if = {
				limit = { has_variable = light_void_infusion }
				var:light_void_infusion >= 0
			}
			trigger_else = {
				NOT = { has_variable = light_void_infusion }
			}
		}
		add = 1
	}
	else = {
		add = -1
	}
}
