light_void_infusion_level_value = {
	value = 0
	if = {
		limit = { is_alive = yes }
		if = {
			limit = { 
				has_trait_xp = { 
					trait = infusion_light
					value >= 100
				}
			}
			add = 3
		}
		else_if = {
			limit = { 
				has_trait_xp = { 
					trait = infusion_light
					value >= 66
				}
			}
			add = 2
		}
		else_if = {
			limit = { 
				has_trait_xp = { 
					trait = infusion_light
					value >= 33
				}
			}
			add = 1
		}
		
		else_if = {
			limit = { 
				has_trait_xp = { 
					trait = infusion_void
					value >= 100
				}
			}
			add = -3
		}
		else_if = {
			limit = { 
				has_trait_xp = { 
					trait = infusion_void
					value >= 66
				}
			}
			add = -2
		}
		else_if = {
			limit = { 
				has_trait_xp = { 
					trait = infusion_void
					value >= 33
				}
			}
			add = -1
		}
	}
}
