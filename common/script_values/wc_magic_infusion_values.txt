light_void_long_bar_stretch_value = {
	value = 0
	
	if = {
		limit = {
			has_variable = light_void_infusion_level
			var:light_void_infusion_level < 0
		}
		add = 200
		subtract = light_void_standard_bar_value
	}
	else = {
		add = light_void_standard_bar_value
	}
}

light_void_standard_bar_value = {
	value = 0
	if = {
		limit = {
			has_variable = light_void_infusion
		}
		add = var:light_void_infusion
		add = 100
	}
}
