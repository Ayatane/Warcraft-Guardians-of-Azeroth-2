mining_story_cycle = {
	on_setup = {
		story_owner = {
			save_profession_effect = {
				PROFESSION = mining
			}
			setup_profession_vars_effect = {
				PROFESSION = flag:mining
			}
		}
	}

	on_end = {
		#remove profession vars
		story_owner = { 
			remove_trait = profession_mining
			remove_variable = Mining_items_made
			remove_variable = Blacksmithing_items_made
			remove_variable = Mining_xp
			every_in_global_list = {
				variable = mining_reagents
				save_scope_as = remove_reagent
				effect_for_scoped_reagent = { REAGENT = scope:remove_reagent EFFECT = remove_known_reagent }
			}
		}
	}

	on_owner_death = {}

	# effect_group = {

	# }
}
