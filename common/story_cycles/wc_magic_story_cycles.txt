story_cycle_planar_fire_spirit = {
	on_setup = {
		story_owner = {
			trigger_event = {
				id = wc_elemental_story_events.001
				days = 3
			}
		}
	}
	on_end = {
		# Adds flags for dynamic locs
		if = {
			limit = {
				NOT = { has_character_flag = planar_fire_spirit }
			}
			add_character_flag = planar_fire_spirit
		}
	}
	on_owner_death = {}
	effect_group = {
		days = 5
		story_owner = {
			trigger_event = {
				id = wc_elemental_story_events.002
			}
		}
	}
	effect_group = {
		days = 10
		story_owner = {
			trigger_event = {
				id = wc_elemental_story_events.003
			}
		}
	}
}
