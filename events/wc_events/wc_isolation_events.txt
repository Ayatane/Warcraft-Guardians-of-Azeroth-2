namespace = isolation

### The cloaking device in Uldum stops working ###
#Pinger event
isolation.1 = {
	scope = none
	hidden = yes
	
	trigger = {
		is_ai = no
		NOT = { has_global_variable = uldum_has_opened_variable }		
		current_date >= 610.1.1
	}
	
	immediate = {
		set_global_variable = uldum_has_opened_variable	
		trigger_event = { id = isolation.2 }
	}
}

#Ping event
isolation.2 = {
	type = character_event
	title = EVTTITLE_isolation_2
	desc = EVTDESC_isolation_2
	theme = diplomacy
	
	override_background = { reference = wc_uldum }
	
	right_portrait = { 
		character = root
		animation = idle
	}
	
	# Time to claim that superweapon I keep hearing rumors about
	option = {
		trigger = {
			is_aggressive_trigger = yes
		}
		name = ISOLATION_2_OPT_A
	}
	# Outsiders inbound
	option = {
		trigger = {
			NOT = {
				is_aggressive_trigger = yes
			}
			capital_county = { title_province = { geographical_region = world_kalimdor_south_uldum } }
		}
		name = ISOLATION_2_OPT_B
	}
	# Time to go exploring
	option = {
		trigger = {
			NOR = {
				is_aggressive_trigger = yes
				capital_county = { title_province = { geographical_region = world_kalimdor_south_uldum } }
			}
		}
		name = ISOLATION_2_OPT_C
	}
}

### MISTS OF PANDARIA ###
# Pinger event
isolation.3 = {
	scope = none
	hidden = yes
	
	trigger = {
		is_ai = no
		NOT = { has_global_variable = mists_of_pandaria_parted_variable }		
		current_date >= 613.1.1
	}
	
	immediate = {
		set_global_variable = mists_of_pandaria_parted_variable	
		trigger_event = { id = isolation.4 }
	}
}

# Announces the parting of the Mists to the world
isolation.4 = {
	type = character_event
	title = EVTTITLE_isolation_4
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					capital_county = { title_province = { geographical_region = world_pandaria } }
				}
				desc = EVTDESC_isolation_4_pandaria
			}
			desc = EVTDESC_isolation_4_azeroth
		}
	}
	theme = diplomacy
	
	override_background = { reference = wc_jade_forest }
	
	right_portrait = { 
		character = root
		animation = idle
	}

	# New lands to explore and conquer!
	option = {
		trigger = {
			is_aggressive_trigger = yes
		}
		name = ISOLATION_4_OPT_A
	}
	# Welcome, wanderers!
	option = {
		trigger = {
			NOT = {
				is_aggressive_trigger = yes
			}
			capital_county = { title_province = { geographical_region = world_pandaria } }
		}
		name = ISOLATION_4_OPT_B
	}
	# Send expeditions
	option = {
		trigger = {
			NOR = {
				is_aggressive_trigger = yes
				capital_county = { title_province = { geographical_region = world_pandaria } }
			}
		}
		name = ISOLATION_4_OPT_C
	}
}
